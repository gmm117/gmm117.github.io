## Syled Components
```javascript
const styled = (newElement) => {
	const el = document.createElement(newElement);
	return (args) => {
      el.style= args[0];
      console.log(args[0]);
      return el;
    }
};

var element = styled('h1')`
	border-radius : 10px;
	color:blue;
`;
console.log(element);
```
